alert(a) //2

var a =1;
function fn1(a){ //有参数，相当于 var a

a =2; //对局部中的a作出更改
}
fn1();
alert(a) //1 全局的a没有被更改

var a =1;
function fn1(a){ //参数被赋值了，相当于 var a =1
alert(a); //1
a =2; //对局部中的a作出更改
}
fn1(a); //参数被赋值了
alert(a) //1 全局的a没有被更改
